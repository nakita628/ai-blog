# Python入門完全ガイド：初心者でも分かるプログラミング言語の基本

[Python](https://www.python.org/)は、読みやすく書きやすいプログラミング言語として世界中で愛用されています。初心者から上級者まで、幅広い層に支持されているPythonの基本を、実際のコード例と共に分かりやすく解説します。

## Pythonとは何か

Pythonは、1991年にGuido van Rossumによって開発されたプログラミング言語です。以下の特徴があります：

- **読みやすい構文**: コードが自然言語に近く、理解しやすい
- **多用途**: Web開発、データ分析、AI、自動化など様々な分野で活用
- **豊富なライブラリ**: 標準ライブラリとサードパーティライブラリが充実
- **クロスプラットフォーム**: Windows、macOS、Linuxで動作
- **オープンソース**: 無料で使用・配布可能

## 関数の定義と使用

### 基本的な関数

```python
# 関数の定義
def greet(name):
    """挨拶をする関数"""
    return f"こんにちは、{name}さん！"

# 関数の呼び出し
message = greet("田中")
print(message)  # こんにちは、田中さん！

# 複数の引数を持つ関数
def calculate_area(length, width):
    """長方形の面積を計算する関数"""
    return length * width

area = calculate_area(5, 3)
print(f"面積: {area}")  # 面積: 15

# デフォルト引数
def greet_with_time(name, time="朝"):
    """時間を指定した挨拶"""
    return f"{time}、{name}さん！"

print(greet_with_time("田中"))           # 朝、田中さん！
print(greet_with_time("田中", "夜"))     # 夜、田中さん！

# 可変長引数
def sum_all(*numbers):
    """任意の数の引数の合計を計算"""
    total = 0
    for num in numbers:
        total += num
    return total

result = sum_all(1, 2, 3, 4, 5)
print(f"合計: {result}")  # 合計: 15
```

### 高度な関数の例

```python
# 再帰関数（フィボナッチ数列）
def fibonacci(n):
    """フィボナッチ数列のn番目の値を計算"""
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# フィボナッチ数列を表示
for i in range(10):
    print(f"fibonacci({i}) = {fibonacci(i)}")

# ラムダ関数（無名関数）
square = lambda x: x ** 2
print(square(5))  # 25

# 高階関数（関数を引数として受け取る）
def apply_operation(numbers, operation):
    """数値のリストに関数を適用"""
    return [operation(x) for x in numbers]

numbers = [1, 2, 3, 4, 5]
doubled = apply_operation(numbers, lambda x: x * 2)
print(doubled)  # [2, 4, 6, 8, 10]
```

## エラーハンドリング

### try-except文

```python
# 基本的なエラーハンドリング
try:
    number = int(input("数値を入力してください: "))
    result = 10 / number
    print(f"結果: {result}")
except ValueError:
    print("数値を入力してください")
except ZeroDivisionError:
    print("ゼロで割ることはできません")
except Exception as e:
    print(f"予期しないエラーが発生しました: {e}")
finally:
    print("処理が完了しました")

# ファイル操作でのエラーハンドリング
def read_file_safely(filename):
    try:
        with open(filename, 'r', encoding='utf-8') as file:
            return file.read()
    except FileNotFoundError:
        print(f"ファイル '{filename}' が見つかりません")
        return None
    except PermissionError:
        print(f"ファイル '{filename}' の読み込み権限がありません")
        return None
    except Exception as e:
        print(f"ファイル読み込み中にエラーが発生しました: {e}")
        return None

# カスタム例外の定義
class CustomError(Exception):
    """カスタム例外クラス"""
    def __init__(self, message):
        self.message = message
        super().__init__(self.message)

def validate_age(age):
    if age < 0:
        raise CustomError("年齢は0以上である必要があります")
    if age > 150:
        raise CustomError("年齢は150以下である必要があります")
    return True

try:
    validate_age(-5)
except CustomError as e:
    print(f"エラー: {e.message}")
```

## クラスとオブジェクト指向

### 基本的なクラス

```python
# 基本的なクラスの定義
class Person:
    """人物を表すクラス"""

    def __init__(self, name, age):
        """コンストラクタ"""
        self.name = name
        self.age = age

    def introduce(self):
        """自己紹介メソッド"""
        return f"私の名前は{self.name}で、{self.age}歳です"

    def have_birthday(self):
        """誕生日を迎えるメソッド"""
        self.age += 1
        print(f"{self.name}さん、{self.age}歳になりました！")

# クラスの使用
person1 = Person("田中太郎", 25)
person2 = Person("佐藤花子", 30)

print(person1.introduce())  # 私の名前は田中太郎で、25歳です
person1.have_birthday()     # 田中太郎さん、26歳になりました！

# 継承の例
class Student(Person):
    """学生クラス（Personクラスを継承）"""

    def __init__(self, name, age, student_id):
        super().__init__(name, age)
        self.student_id = student_id
        self.grades = []

    def add_grade(self, grade):
        """成績を追加"""
        self.grades.append(grade)

    def get_average_grade(self):
        """平均点を計算"""
        if not self.grades:
            return 0
        return sum(self.grades) / len(self.grades)

    def introduce(self):
        """自己紹介メソッドをオーバーライド"""
        base_intro = super().introduce()
        return f"{base_intro}、学籍番号は{self.student_id}です"

# 継承クラスの使用
student = Student("鈴木一郎", 20, "2024001")
student.add_grade(85)
student.add_grade(92)
student.add_grade(78)

print(student.introduce())
print(f"平均点: {student.get_average_grade():.1f}")
```

## 実践的なプログラム例

### 簡単な計算機

```python
class Calculator:
    """簡単な計算機クラス"""

    def __init__(self):
        self.history = []

    def add(self, a, b):
        result = a + b
        self.history.append(f"{a} + {b} = {result}")
        return result

    def subtract(self, a, b):
        result = a - b
        self.history.append(f"{a} - {b} = {result}")
        return result

    def multiply(self, a, b):
        result = a * b
        self.history.append(f"{a} × {b} = {result}")
        return result

    def divide(self, a, b):
        if b == 0:
            raise ValueError("ゼロで割ることはできません")
        result = a / b
        self.history.append(f"{a} ÷ {b} = {result}")
        return result

    def show_history(self):
        print("計算履歴:")
        for calculation in self.history:
            print(f"  {calculation}")

# 計算機の使用例
calc = Calculator()

try:
    print(f"5 + 3 = {calc.add(5, 3)}")
    print(f"10 - 4 = {calc.subtract(10, 4)}")
    print(f"6 × 7 = {calc.multiply(6, 7)}")
    print(f"15 ÷ 3 = {calc.divide(15, 3)}")

    calc.show_history()

except ValueError as e:
    print(f"エラー: {e}")
```

### 単語カウンター

```python
def count_words(text):
    """テキスト内の単語数をカウント"""
    words = text.split()
    word_count = {}

    for word in words:
        # 句読点を除去
        clean_word = word.strip('.,!?;:"()[]{}')
        clean_word = clean_word.lower()

        if clean_word:
            word_count[clean_word] = word_count.get(clean_word, 0) + 1

    return word_count

def display_word_count(word_count):
    """単語数の結果を表示"""
    print("単語数ランキング:")
    sorted_words = sorted(word_count.items(), key=lambda x: x[1], reverse=True)

    for word, count in sorted_words[:10]:  # 上位10位まで表示
        print(f"  {word}: {count}回")

# 使用例
sample_text = """
Pythonは素晴らしいプログラミング言語です。
Pythonは読みやすく、書きやすいです。
Pythonは多くの分野で使われています。
"""

word_count = count_words(sample_text)
display_word_count(word_count)
```
