Pinia完全入門：Vue.jsの直感的な状態管理ライブラリでアプリケーションを構築する ​

Piniaは、Vue.jsの公式状態管理ライブラリで、Vuexの後継として設計されました。型安全性、直感的なAPI、優れた開発者体験を提供し、Vue.jsアプリケーションでの効率的な状態管理を実現します。この記事では、Piniaの基本概念から実装方法まで、実践的なコード例を通じて詳しく解説します。

Piniaとは ​

Piniaは、Vue.jsアプリケーションのための直感的で型安全な状態管理ライブラリです。以下の特徴を持っています：

  • 直感的なAPI: コンポーネントのように親しみやすいストア設計
  • 型安全性: TypeScriptの型推論による自動補完
  • DevTools対応: Vue DevToolsとの完全統合
  • モジュラー設計: 複数ストアの自動コード分割
  • 軽量: 約1.5KBの軽量サイズ
  • 拡張性: プラグインによる機能拡張

// stores/counter.ts
import { defineStore } from 'pinia';

export const useCounterStore = defineStore('counter', {
  // 状態（state）
  state: () => ({
    count: 0,
    name: 'Counter Store'
  }),

  // ゲッター（getters）
  getters: {
    doubleCount: (state) => state.count * 2,
    doubleCountPlusOne: (state) => state.doubleCount + 1,
    // 他のゲッターを参照
    greeting: (state) => `Hello, ${state.name}! Count is ${state.count}`
  },

  // アクション（actions）
  actions: {
    increment() {
      this.count++;
    },
    decrement() {
      this.count--;
    },
    incrementBy(amount: number) {
      this.count += amount;
    },
    reset() {
      this.count = 0;
    },
    updateName(newName: string) {
      this.name = newName;
    }
  }
});

ユーザー管理ストア ​

typescript
// stores/user.ts
import { defineStore } from 'pinia';

export interface User {
  id: string;
  name: string;
  email: string;
  avatar?: string;
  createdAt: Date;
}

export interface UserState {
  users: User[];
  currentUser: User | null;
  loading: boolean;
  error: string | null;
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    users: [],
    currentUser: null,
    loading: false,
    error: null
  }),

  getters: {
    // 全ユーザー数
    userCount: (state) => state.users.length,
    
    // アクティブユーザー（現在のユーザー）
    isLoggedIn: (state) => state.currentUser !== null,
    
    // ユーザー検索
    getUserById: (state) => (id: string) => 
      state.users.find(user => user.id === id),
    
    // ユーザー名で検索
    getUsersByName: (state) => (name: string) =>
      state.users.filter(user => 
        user.name.toLowerCase().includes(name.toLowerCase())
      ),
    
    // 最近作成されたユーザー
    recentUsers: (state) => 
      [...state.users]
        .sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime())
        .slice(0, 5)
  },

  actions: {
    // ユーザー一覧を取得
    async fetchUsers() {
      this.loading = true;
      this.error = null;
      
      try {
        // 実際のAPI呼び出し
        const response = await fetch('/api/users');
        if (!response.ok) {
          throw new Error('Failed to fetch users');
        }
        
        const users = await response.json();
        this.users = users.map((user: any) => ({
          ...user,
          createdAt: new Date(user.createdAt)
        }));
      } catch (error) {
        this.error = error instanceof Error ? error.message : 'Unknown error';
      } finally {
        this.loading = false;
      }
    },

    // ユーザーを作成
    async createUser(userData: Omit<User, 'id' | 'createdAt'>) {
      this.loading = true;
      this.error = null;
      
      try {
        const response = await fetch('/api/users', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(userData)
        });
        
        if (!response.ok) {
          throw new Error('Failed to create user');
        }
        
        const newUser = await response.json();
        this.users.push({
          ...newUser,
          createdAt: new Date(newUser.createdAt)
        });
        
        return newUser;
      } catch (error) {
        this.error = error instanceof Error ? error.message : 'Unknown error';
        throw error;
      } finally {
        this.loading = false;
      }
    },

    // ユーザーを更新
    async updateUser(id: string, userData: Partial<User>) {
      this.loading = true;
      this.error = null;
      
      try {
        const response = await fetch(`/api/users/${id}`, {
          method: 'PUT',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(userData)
        });
        
        if (!response.ok) {
          throw new Error('Failed to update user');
        }
        
        const updatedUser = await response.json();
        const index = this.users.findIndex(user => user.id === id);
        if (index !== -1) {
          this.users[index] = {
            ...updatedUser,
            createdAt: new Date(updatedUser.createdAt)
          };
        }
        
        return updatedUser;
      } catch (error) {
        this.error = error instanceof Error ? error.message : 'Unknown error';
        throw error;
      } finally {
        this.loading = false;
      }
    },

    // ユーザーを削除
    async deleteUser(id: string) {
      this.loading = true;
      this.error = null;
      
      try {
        const response = await fetch(`/api/users/${id}`, {
          method: 'DELETE'
        });
        
        if (!response.ok) {
          throw new Error('Failed to delete user');
        }
        
        this.users = this.users.filter(user => user.id !== id);
        
        // 現在のユーザーが削除された場合
        if (this.currentUser?.id === id) {
          this.currentUser = null;
        }
      } catch (error) {
        this.error = error instanceof Error ? error.message : 'Unknown error';
        throw error;
      } finally {
        this.loading = false;
      }
    },

    // ログイン
    login(user: User) {
      this.currentUser = user;
    },

    // ログアウト
    logout() {
      this.currentUser = null;
    },

    // エラーをクリア
    clearError() {
      this.error = null;
    }
  }
});

ショッピングカートストア ​

typescript
// stores/cart.ts
import { defineStore } from 'pinia';
import { useUserStore } from './user';

export interface CartItem {
  id: string;
  name: string;
  price: number;
  quantity: number;
  image?: string;
}

export const useCartStore = defineStore('cart', {
  state: () => ({
    items: [] as CartItem[],
    isOpen: false
  }),

  getters: {
    // カート内のアイテム数
    itemCount: (state) => 
      state.items.reduce((total, item) => total + item.quantity, 0),
    
    // 合計金額
    totalPrice: (state) => 
      state.items.reduce((total, item) => total + (item.price * item.quantity), 0),
    
    // カートが空かどうか
    isEmpty: (state) => state.items.length === 0,
    
    // 特定のアイテムがカートにあるかどうか
    hasItem: (state) => (itemId: string) => 
      state.items.some(item => item.id === itemId)
  },

  actions: {
    // アイテムを追加
    addItem(item: Omit<CartItem, 'quantity'>) {
      const existingItem = this.items.find(i => i.id === item.id);
      
      if (existingItem) {
        existingItem.quantity++;
      } else {
        this.items.push({ ...item, quantity: 1 });
      }
    },

    // アイテムを削除
    removeItem(itemId: string) {
      this.items = this.items.filter(item => item.id !== itemId);
    },

    // 数量を更新
    updateQuantity(itemId: string, quantity: number) {
      const item = this.items.find(i => i.id === itemId);
      if (item) {
        if (quantity <= 0) {
          this.removeItem(itemId);
        } else {
          item.quantity = quantity;
        }
      }
    },

    // カートをクリア
    clearCart() {
      this.items = [];
    },

    // カートを開く/閉じる
    toggleCart() {
      this.isOpen = !this.isOpen;
    },

    // チェックアウト
    async checkout() {
      const userStore = useUserStore();
      
      if (!userStore.isLoggedIn) {
        throw new Error('ログインが必要です');
      }
      
      try {
        const response = await fetch('/api/checkout', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${userStore.currentUser?.id}`
          },
          body: JSON.stringify({
            items: this.items,
            userId: userStore.currentUser?.id
          })
        });
        
        if (!response.ok) {
          throw new Error('チェックアウトに失敗しました');
        }
        
        // チェックアウト成功後、カートをクリア
        this.clearCart();
        this.isOpen = false;
        
        return await response.json();
      } catch (error) {
        console.error('Checkout error:', error);
        throw error;
      }
    }
  }
});

商品ストア ​

// stores/product.ts
import { defineStore } from 'pinia';

export interface Product {
  id: string;
  name: string;
  price: number;
  description: string;
  image: string;
  category: string;
  inStock: boolean;
  rating: number;
}

export const useProductStore = defineStore('product', {
  state: () => ({
    products: [] as Product[],
    loading: false,
    error: null as string | null,
    searchQuery: '',
    selectedCategory: 'all',
    sortBy: 'name' as 'name' | 'price' | 'rating'
  }),

  getters: {
    // フィルタリングされた商品
    filteredProducts: (state) => {
      let filtered = state.products;
      
      // カテゴリでフィルタリング
      if (state.selectedCategory !== 'all') {
        filtered = filtered.filter(product => product.category === state.selectedCategory);
      }
      
      // 検索クエリでフィルタリング
      if (state.searchQuery) {
        const query = state.searchQuery.toLowerCase();
        filtered = filtered.filter(product => 
          product.name.toLowerCase().includes(query) ||
          product.description.toLowerCase().includes(query)
        );
      }
      
      // ソート
      return filtered.sort((a, b) => {
        switch (state.sortBy) {
          case 'price':
            return a.price - b.price;
          case 'rating':
            return b.rating - a.rating;
          default:
            return a.name.localeCompare(b.name);
        }
      });
    },
    
    // カテゴリ一覧
    categories: (state) => {
      const categories = new Set(state.products.map(product => product.category));
      return Array.from(categories);
    },
    
    // 在庫切れ商品
    outOfStockProducts: (state) => 
      state.products.filter(product => !product.inStock)
  },

  actions: {
    // 商品一覧を取得
    async fetchProducts() {
      this.loading = true;
      this.error = null;
      
      try {
        const response = await fetch('/api/products');
        if (!response.ok) {
          throw new Error('商品の取得に失敗しました');
        }
        
        this.products = await response.json();
      } catch (error) {
        this.error = error instanceof Error ? error.message : 'Unknown error';
      } finally {
        this.loading = false;
      }
    },

    // 検索クエリを設定
    setSearchQuery(query: string) {
      this.searchQuery = query;
    },

    // カテゴリを設定
    setCategory(category: string) {
      this.selectedCategory = category;
    },

    // ソート方法を設定
    setSortBy(sortBy: 'name' | 'price' | 'rating') {
      this.sortBy = sortBy;
    }
  }
});

永続化プラグイン ​

typescript
// plugins/persistence.ts
import { PiniaPluginContext } from 'pinia';

export function persistencePlugin({ store }: PiniaPluginContext) {
  // ストアの状態をローカルストレージに保存
  const storageKey = `pinia-${store.$id}`;
  
  // 初期化時にローカルストレージから復元
  const stored = localStorage.getItem(storageKey);
  if (stored) {
    store.$patch(JSON.parse(stored));
  }
  
  // 状態変更時にローカルストレージに保存
  store.$subscribe((mutation, state) => {
    localStorage.setItem(storageKey, JSON.stringify(state));
  });
}

// main.ts
import { createPinia } from 'pinia';
import { persistencePlugin } from './plugins/persistence';

const pinia = createPinia();
pinia.use(persistencePlugin);

// plugins/logger.ts
import { PiniaPluginContext } from 'pinia';

export function loggerPlugin({ store }: PiniaPluginContext) {
  // アクション実行前のログ
  store.$onAction(({ name, args, after, onError }) => {
    console.log(`Action ${name} started with args:`, args);
    
    after((result) => {
      console.log(`Action ${name} completed with result:`, result);
    });
    
    onError((error) => {
      console.error(`Action ${name} failed with error:`, error);
    });
  });
  
  // 状態変更のログ
  store.$subscribe((mutation, state) => {
    console.log(`State changed in ${store.$id}:`, mutation, state);
  });
}

// stores/typed-store.ts
import { defineStore } from 'pinia';

interface User {
  id: string;
  name: string;
  email: string;
}

interface UserState {
  users: User[];
  currentUser: User | null;
  loading: boolean;

export const useTypedStore = defineStore('typed', {
  state: (): UserState => ({
    users: [],
    currentUser: null,
    loading: false
  }),

  getters: {
    // 型安全なゲッター
    userCount: (state): number => state.users.length,
    
    // パラメータ付きゲッター
    getUserById: (state) => (id: string): User | undefined => 
      state.users.find(user => user.id === id),
    
    // 他のゲッターを参照

  actions: {
    // 型安全なアクション
    async fetchUsers(): Promise<void> {
      this.loading = true;
      try {
        const response = await fetch('/api/users');
        this.users = await response.json();
      } finally {
        this.loading = false;
      }
    },

        headers: { 'Content-Type': 'application/json' },

      });
      
      const newUser = await response.json();
      this.users.push(newUser);
      return newUser;
    },

    // 戻り値のあるアクション
    login(email: string, password: string): boolean {
      const user = this.users.find(u => u.email === email);
      if (user) {
        this.currentUser = user;
        return true;
      }
      return false;
    }
  }
});

まとめ ​

Piniaは、Vue.jsアプリケーションの状態管理を効率化する強力なライブラリです。主なポイントは以下の通りです：

主要な特徴 ​

  • 直感的なAPI: コンポーネントのように親しみやすい設計
  • 型安全性: TypeScriptの完全サポート
  • DevTools対応: Vue DevToolsとの完全統合
  • 軽量: 約1.5KBの軽量サイズ
  • 拡張性: プラグインによる機能拡張

実装のポイント ​

  • 適切なストア設計: 単一責任の原則に従ったストア設計
  • 型安全性: TypeScriptの型システムを活用
  • ストア間の連携: 適切な依存関係の管理
  • プラグインの活用: 永続化やログなどの機能拡張
